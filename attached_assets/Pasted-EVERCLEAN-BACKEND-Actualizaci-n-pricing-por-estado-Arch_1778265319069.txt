app.post('/api/bookings/:id/eta', authenticateToken, async (req, res) => {
  try {
    const { eta_minutes, eta_message, pro_lat, pro_lng } = req.body;
    const booking = await prisma.bookings.update({
      where: { id: req.params.id },
      data: {
        eta_minutes: eta_minutes ?? null,
        eta_message: eta_message ?? null,
        pro_lat:     pro_lat     ?? null,
        pro_lng:     pro_lng     ?? null,
        eta_sent_at: new Date(),
      },
    });
    res.json({ success: true, booking });
  } catch (err) {
    console.error('POST /bookings/:id/eta error:', err);
    res.status(500).json({ error: err.message });
  }
});

app.patch('/api/companies/me', authenticateToken, async (req, res) => {
  try {
    const userId = req.user.id;
    const { name, full_name, phone, billingAddress, billing_address,
            address, billingCity, billing_city, city, billingState,
            billing_state, state, billingZip, billing_zip, zip_code } = req.body;

    const fullName = name || full_name;
    const addr = billingAddress || billing_address || address;
    const cty  = billingCity    || billing_city    || city;
    const st   = billingState   || billing_state   || state;
    const zip  = billingZip     || billing_zip     || zip_code;

    if (fullName || phone) {
      const d = {};
      if (fullName) d.full_name = fullName;
      if (phone)    d.phone     = phone;
      await prisma.users.update({ where: { id: userId }, data: d });
    }

    const cd = {};
    if (fullName) cd.name            = fullName;
    if (addr)     cd.billing_address = addr;
    if (cty)      cd.billing_city    = cty;
    if (st)       cd.billing_state   = st;
    if (zip)      cd.billing_zip     = zip;

    if (Object.keys(cd).length > 0) {
      await prisma.companies.upsert({
        where:  { user_id: userId },
        update: cd,
        create: { user_id: userId, ...cd, name: fullName || 'Client' },
      });
    }
    res.json({ success: true });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});
